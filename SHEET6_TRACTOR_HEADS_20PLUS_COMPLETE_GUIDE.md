class TaxCalculatorTractorHeads20Plus {
  final double cifUSD;
  final double exchangeRate;

  double get customsValue => cifUSD * exchangeRate;

  // Row 13: Stamp Duty
  double get stampDuty => 18000.0;
  
  // Row 14: Form Fees
  double get formFees => 35000.0;
  
  // Row 15: Infrastructure Levy - ZERO!
  double get infrastructureLevy => 0.0;  // ✅✅ 0%!
  
  // Row 16: TOTAL TAXES
  double get totalTaxes {
    return importDuty  // 0
         + vat
         + wht
         + environmentalLevy
         + registrationFees
         + stampDuty
         + formFees
         + infrastructureLevy;  // 0
  }
  
  // Calculate savings vs other categories
  double getSavingsVsLightTruck(double cifUSD, double rate) {
    double customsVal = cifUSD * rate;
    
    // Light truck would pay:
    double lightTruckImportDuty = customsVal * 0.20;
    double lightTruckInfrastructure = customsVal * 0.015;
    double lightTruckRegFees = 1500000 - 1250000;
    
    return lightTruckImportDuty + lightTruckInfrastructure + lightTruckRegFees;
  }
  
  Map<String, double> getTaxBreakdown() {
    return {
      'Import Duty (0%)': importDuty,  // ZERO!
      'V.A.T (18%)': vat,
      'W.H.T (6%)': wht,
      'Environmental Levy (20%)': environmentalLevy,
      'Registration Fees': registrationFees,
      'Stamp Duty': stampDuty,
      'Form Fees': formFees,
      'Infrastructure Levy (0%)': infrastructureLevy,  // ZERO!
      'TOTAL': totalTaxes,
    };
  }
}
